   * @memberof SRefData
   * @returns {Promise<SRefData>}
   */
  async getWGS84() {
    try {
      var { SRefDataId } = await SR.getWGS84(this._MGSRefDataId);
      var sRefData = new SRefData();
      sRefData._MGSRefDataId = SRefDataId;
      return sRefData;
    } catch (e) {
      console.error(e);
    }
  }
