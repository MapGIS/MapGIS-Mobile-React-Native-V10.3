import { NativeModules } from "react-native";
let LS = NativeModules.JSLineStyle;

/**
 * @class LineStyle
 * @description 线样式（实线）
 */
export default class LineStyle{

    /**
     * 构造一个新的LineStyle对象，可通过无参或有参构造。
     * 有参构造的参数为：颜色（int类型的Number）、线宽（float类型的Number）
     * 
     * @memberof LineStyle
     * @returns {Promise<LineStyle>}
     */
    async createObj(){
        try {
            if(typeof arguments[0] === "number" && typeof arguments[1] === "number"){
                 var {LineStyleId} = await LS.createObjByCW(arguments[0], arguments[1]);
                 var lineStyle = new LineStyle();
                 lineStyle._MGLineStyleId = LineStyleId;
                 return lineStyle;
            }else{
                var {LineStyleId} = await LS.createObj();
                var lineStyle = new LineStyle();
                lineStyle._MGLineStyleId = LineStyleId;
                return lineStyle;
            }
        } catch (e) {
            console.error(e);
        }
    }
